# Phase 4.2 — Warning routing to History + Settings Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Surface the existing 20 `WarningId` catalog on History (top strip, 5 IDs) and Settings ("Permissions & status" section, 7 IDs) reusing the shipped v3 chrome, with multi-active surfacing and hybrid dismissibility (History session-local; Settings non-dismissable; Record unchanged).

**Architecture:** Extract a new `WarningPrecedence.allActive(...)` that emits the full ordinal-sorted set of currently-active IDs; refactor the existing `resolve(...)` to call it via `firstOrNull()` (single condition-wiring source, Record invariant preserved). Add `WarningCenterViewModel.activeWarningsFor(screen)` deriving from `allActive(...)` filtered by per-screen allowlist + snoozed set + History per-session dismiss set with signal-down-edge clearing.

```kotlin
fun resolve(
    cameraPermissionGranted: Boolean = true,
    exactAlarmGranted: Boolean = true,
    storageInsufficient: Boolean = false,
    thermal: ThermalStatus = ThermalStatus.NONE,
    power: PowerState = PowerState.OK,

    batteryOptimizationExempt: Boolean = true,
    storageLowMidRec: Boolean = false,
    autoStopEcho: TerminalEcho? = null,
    cantMergeActive: Boolean = false,
```

```kotlin
/**
 * Phase 4.2 — emit every currently-active [WarningId] ordinal-sorted.
 * The single condition-wiring source; [resolve] returns `firstOrNull()`
 * of this list. Pure function — no flow plumbing.
 */
fun allActive(
    cameraPermissionGranted: Boolean = true,
    exactAlarmGranted: Boolean = true,
    storageInsufficient: Boolean = false,
    thermal: ThermalStatus = ThermalStatus.NONE,
    power: PowerState = PowerState.OK,
    camera: CameraSignalState = CameraSignalState.READY,
    microphonePermissionGranted: Boolean = true,
    notificationsGranted: Boolean = true,
    batteryOptimizationExempt: Boolean = true,
    storageLowMidRec: Boolean = false,
    autoStopEcho: TerminalEcho? = null,
    cantMergeActive: Boolean = false,
): List<WarningId> {
    val out = mutableListOf<WarningId>()
    if (!cameraPermissionGranted) out += WarningId.CAMERA_PERMISSION_DENIED
    if (!exactAlarmGranted) out += WarningId.EXACT_ALARM_DENIED
    if (storageInsufficient) out += WarningId.STORAGE_INSUFFICIENT
    if (thermal == ThermalStatus.SHUTDOWN) out += WarningId.THERMAL_SHUTDOWN
    if (thermal == ThermalStatus.EMERGENCY) out += WarningId.THERMAL_EMERGENCY
    if (thermal == ThermalStatus.CRITICAL) out += WarningId.THERMAL_CRITICAL
    if (power == PowerState.BATTERY_CRITICAL) out += WarningId.BATTERY_CRITICAL
    if (camera == CameraSignalState.IN_USE) out += WarningId.CAMERA_IN_USE
    if (camera == CameraSignalState.DISABLED) out += WarningId.CAMERA_DISABLED
    if (power == PowerState.BATTERY_LOW) out += WarningId.BATTERY_LOW
    if (storageLowMidRec) out += WarningId.STORAGE_LOW_MID_REC
    when (autoStopEcho?.stopReason) {
        StopReason.LOW_STORAGE -> out += WarningId.STORAGE_FULL_AUTOSTOPPED
        StopReason.THERMAL -> out += WarningId.THERMAL_AUTOSTOPPED
        else -> Unit
    }
    if (cantMergeActive) out += WarningId.CANT_MERGE
    if (thermal == ThermalStatus.SEVERE) out += WarningId.THERMAL_SEVERE
    if (!microphonePermissionGranted) out += WarningId.MICROPHONE_DENIED
    if (!batteryOptimizationExempt) out += WarningId.BATTERY_OPTIMIZATION_ON
    if (power == PowerState.POWER_SAVE) out += WarningId.POWER_SAVE_MODE
    if (thermal == ThermalStatus.MODERATE) out += WarningId.THERMAL_MODERATE
    if (!notificationsGranted) out += WarningId.NOTIFICATIONS_DENIED
    return out
}
```

```kotlin
// Inside WarningCenterViewModel — after the existing _resolvedWarning derivation:

private val _allActive: StateFlow<List<WarningId>> = aggregateAllActive(
    cameraPermissionGranted, exactAlarmGranted, storageInsufficient,
    thermal, power, camera, microphonePermissionGranted,
    notificationsGranted, batteryOptimizationExempt, storageLowMidRec,
    autoStopEcho, _cantMergeActive,
).stateIn(scope, SharingStarted.Eagerly, emptyList())

/**
 * Phase 4.2 — per-screen multi-active flow. Returns the ordinal-sorted
 * list of currently-active WarningIds filtered to the screen's allowlist,
 * minus any IDs the user has snoozed forever. Record always returns
 * empty — Record screen consumes [activeWarning] (single resolved ID)
 * which is the existing hard-invariant API.
 */
fun activeWarningsFor(screen: WarningScreen): StateFlow<List<WarningId>> = when (screen) {
    WarningScreen.Record -> MutableStateFlow<List<WarningId>>(emptyList()).asStateFlow()
    WarningScreen.History -> _historyActive
    WarningScreen.Settings -> _settingsActive
}

private val _historyActive: StateFlow<List<WarningId>> = combine(
    _allActive, _snoozedForever,
) { all, snoozed ->
    all.filter { it in HISTORY_WARNINGS && it !in snoozed }
}.stateIn(scope, SharingStarted.Eagerly, emptyList())

private val _settingsActive: StateFlow<List<WarningId>> = combine(
    _allActive, _snoozedForever,
) { all, snoozed ->
    all.filter { it in SETTINGS_WARNINGS && it !in snoozed }
}.stateIn(scope, SharingStarted.Eagerly, emptyList())
```
